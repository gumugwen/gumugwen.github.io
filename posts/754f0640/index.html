    3.hadoop自动化部署脚本文件

    大数据 docker
    发布时间 : 2021-11-06 00:00

    使用教程

    使用前提docker+hadoop_node.tar.gz+JDK

    1.把load.sh和master.sh和slave1.sh、slave2.sh、slave3.sh上传到项目里

    2.进入/mnt/cgshare目录

    3.执行./load.sh

    4.打开另外三个窗口并进入容器 

    docker exec -it --privileged slave1 /bin/bash
docker exec -it --privileged slave2 /bin/bash
docker exec -it --privileged slave3 /bin/bash

    5.执行./master.sh 出现 密码输入密码83953588abc 出现yes输入yes

    6.在另外三个窗口 执行./slave1.sh、./slave2.sh、./slave3.sh

    7.所有容器执行一遍 source ~/.bashrc

    在master执行如下语句

    hdfs namenode -format  (只首次需要！！！)|
start-dfs.sh
start-yarn.sh
mr-jobhistory-daemon.sh start historyserver

    代码

    load.sh

    

docker load < /cg/images/hadoop_node.tar.gz

docker run --name master --privileged --hostname master --ip 172.18.0.2 --add-host=slave1:172.18.0.3  --add-host=slave2:172.18.0.4 --add-host=slave3:172.18.0.5 -itd -v /cgsrc:/cgsrc:ro -v /headless/course/:/course hadoop_node /service_start.sh

docker run --name slave1 --privileged --hostname slave1 --ip 172.18.0.3 --add-host=master:172.18.0.2  --add-host=slave2:172.18.0.4 --add-host=slave3:172.18.0.5  -itd -v /cgsrc:/cgsrc:ro hadoop_node /service_start.sh

docker run --name slave2 --privileged --hostname slave2 --ip 172.18.0.4 --add-host=master:172.18.0.2 --add-host=slave1:172.18.0.3  --add-host=slave3:172.18.0.5 -itd -v /cgsrc:/cgsrc:ro hadoop_node /service_start.sh

docker run --name slave3 --privileged --hostname slave3 --ip 172.18.0.5 --add-host=master:172.18.0.2 --add-host=slave1:172.18.0.3  --add-host=slave2:172.18.0.4 -itd -v /cgsrc:/cgsrc:ro hadoop_node /service_start.sh

docker start master
docker start slave1
docker start slave2
docker start slave3

docker cp  master.sh master:/
docker cp  slave1.sh slave1:/
docker cp  slave2.sh slave2:/
docker cp  slave3.sh slave3:/

echo "docker exec -it --privileged slave1 /bin/bash"
echo "docker exec -it --privileged slave2 /bin/bash"
echo "docker exec -it --privileged slave3 /bin/bash"

docker exec -it --privileged master /bin/bash

    master.sh

    mkdir /usr/local/java
cp /cgsrc/jdk-8u171-linux-x64.tar.gz /usr/local/java/
cd /usr/local/java/
tar -zxvf jdk-8u171-linux-x64.tar.gz
rm -f jdk-8u171-linux-x64.tar.gz
echo 'export JAVA_HOME=/usr/local/java/jdk1.8.0_171
export CLASSPATH=.:${JAVA_HOME}/jre/lib/rt.jar:${JAVA_HOME}/lib/dt.jar:${JAVA_HOME}/lib/tools.jar
export PATH=$PATH:${JAVA_HOME}/bin'>>~/.bashrc

ping master -c 3
ping slave1 -c 3
ping slave2 -c 3
ping slave3 -c 3
echo "-----------网络目前已经联通--------------"

echo "-----------配置无密码登录--------------"
echo "-----------下面需要三个回车  --------------"
cd ~/.ssh
rm -f ./id_rsa*
ssh-keygen -t rsa

cat ./id_rsa.pub >> ./authorized_keys
echo "-----------下面需要输入密码 83953588abc  --------------"

scp ~/.ssh/id_rsa.pub root@slave1:/root
scp ~/.ssh/id_rsa.pub root@slave2:/root
scp ~/.ssh/id_rsa.pub root@slave3:/root

ssh-keyscan localhost >> ~/.ssh/known_hosts

cp /cgsrc/hadoop-2.7.1.tar.gz /usr/local/
cd /usr/local
tar -zxvf hadoop-2.7.1.tar.gz
rm -f hadoop-2.7.1.tar.gz

mv hadoop-2.7.1/ hadoop

echo 'export PATH=$PATH:/usr/local/hadoop/bin:/usr/local/hadoop/sbin'>>~/.bashrc
source ~/.bashrc

echo $JAVA_HOME
java -version
echo "-----------java环境配置成功-------------"
hadoop version
echo "-----------hadoop安装成功--------------"

cd /usr/local/hadoop/etc/hadoop

echo 'slave1
slave2
slave3'>slaves

echo "设置slaves成功->>"
cat slaves

echo '<configuration>
    <property>
        <name>fs.defaultFS</name>
        <value>hdfs://master:9000</value>
    </property>
    <property>
        <name>hadoop.tmp.dir</name>
        <value>file:/usr/local/hadoop/tmp</value>
    </property>
</configuration>'>core-site.xml

echo '<configuration>
    <property>
        <name>dfs.namenode.secondary.http-address</name>
        <value>master:50090</value>
    </property>
    <property>
        <name>dfs.replication</name>
        <value>3</value>
    </property>
    <property>
        <name>dfs.namenode.name.dir</name>
        <value>file:/usr/local/hadoop/tmp/dfs/name</value>
    </property>
    <property>
        <name>dfs.datanode.data.dir</name>
        <value>file:/usr/local/hadoop/tmp/dfs/date</value>
    </property>
</configuration>'>hdfs-site.xml

cp mapred-site.xml.template mapred-site.xml

echo '<configuration>
    <property>
        <name>mapreduce.framework.name</name>
        <value>yarn</value>
    </property>
    <property>
        <name>mapreduce.jobhistory.address</name>
        <value>master:10020</value>
    </property>
    <property>
        <name>mapreduce.jobhistory.webapp.address</name>
        <value>master:19888</value>
    </property>
</configuration>'>mapred-site.xml

echo '<configuration>
    <property>
        <name>yarn.resourcemanager.hostname</name>
        <value>master</value>
    </property>
    <property>
        <name>yarn.nodemanager.aux-services</name>
        <value>mapreduce_shuffle</value>
    </property>
    <property>
        <name>yarn.nodemanager.disk-health-checker.max-disk-utilization-per-disk-percentage</name>
        <value>98.5</value>
    </property>
</configuration>'>yarn-site.xml
echo "-------------hadoop配置成功-----------------"

cd /usr/local
rm -rf ./hadoop/tmp                         #删除临时文件
rm -rf ./hadoop/logs/*                      #删除日志文件

tar -zcf ~/hadoop.master.tar.gz ./hadoop
cd ~
scp ./hadoop.master.tar.gz root@slave1:/root
scp ./hadoop.master.tar.gz root@slave2:/root
scp ./hadoop.master.tar.gz root@slave3:/root

echo "-------------hadoop全部配置完毕，需要你在master启动
source ~/.bashrc
hdfs namenode -format  (只首次需要！！！)|
start-dfs.sh
start-yarn.sh
