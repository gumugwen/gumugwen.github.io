    12、74HC595一位数码管0~9显示

    const int datapin = 23;         //DS
const int clockpin = 18;        //SH_CP
const int latchpin = 5;         //ST_CP
const byte num[] = {
  0b11111100,                     //0
  0b01100000,                     //1
  0b11011010,                     //2
  0b11110010,                     //3
  0b01100110,                     //4
  0b10110110,                     //5
  0b10111110,                     //6
  0b11100000,                     //7
  0b11111110,                     //8
  0b11110110,                     //9
  0b00000001                      //.
};
void digitalTube_1(byte data_num)
{
  unsigned char i;
  for(i = 0;i<8;i++)
    {
      digitalWrite(datapin,!bitRead(num[data_num],i));   
      //读取data_num的每一个位，并置位
      digitalWrite(clockpin,HIGH);         //clk 引脚上升沿，寄存器移位
      digitalWrite(clockpin,LOW);         
    }  
    digitalWrite(latchpin,HIGH);           //更新数据
    digitalWrite(latchpin,LOW);  
}
void setup() {
  // put your setup code here, to run once:
  pinMode(datapin,OUTPUT);
  pinMode(clockpin,OUTPUT);
  pinMode(latchpin,OUTPUT);         //配置为输出模式
  digitalWrite(clockpin,LOW);       //设置为低电平状态
  digitalWrite(latchpin,LOW);       //设置为低电平状态
}
void loop() {
  // put your main code here, to run repeatedly:
  for(int j = 0;j<=10;j++)     //0~9显示
  {
    digitalTube_1(j);
    delay(300);
  }
}

    14、按键消抖

    const int keypin = 12;         //按键
const int ledpin = 2 ;        //LED灯
int preKeyVal = 1;             //保持先前按键值
int preKeyState = 1;           //保持按键状态值
boolean KeyFlag = false;       //按键状态变量
unsigned int preTime = 0;      //保存millis()返回值
int debounceDelay = 10;        //设定消抖时间间隔
void getKeyState(void)
{
  int keyVal = digitalRead(keypin) ;    //读取按键值
  if(keyVal!=preKeyVal)
  {
    preTime = millis();                 //获取当前时间
    preKeyVal = keyVal;                 //将当前值赋值给先前值
  } 
  if((millis() - preTime)>debounceDelay)  //判断时间间隔是否大于设定的消抖时间，再判断按键状态是否发生变化
  {
    if(keyVal!=preKeyState)
    {
      preKeyState = keyVal;
      if(keyVal==HIGH)
      {
        KeyFlag = !KeyFlag;  
      }  
    } 
  }
}
void setup() {
  // put your setup code here, to run once:
  pinMode(keypin,INPUT);
  pinMode(ledpin,OUTPUT);
}
void loop() {
  // put your main code here, to run repeatedly:
  getKeyState();
  digitalWrite(ledpin,KeyFlag);
}
